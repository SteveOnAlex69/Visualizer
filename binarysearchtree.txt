struct Node{
    int val;
    Node *childL, *childR;
    Node(int x = 0){val = x; childL = childR = nullptr;}
};  

struct BinarySearchTree{
    Node* root;

    BinarySearchTree(){
        root = nullptr;
    }

    void insert(int x){
        insert(root, x);
    }

    void insert(Node* &root, int x){
        if (root == nullptr) {
            root = new Node(x);
            return;
        }
        if (x < root -> val) insert(root -> childL, x);
        else insert(root -> childR, x);
    }

    void traverse(){
        vector<int> ans;
        traverse(root, ans);
        printArr(ans);
    }

    void traverse(Node *root, vector<int> &ans){
        if (root == nullptr) return;
        traverse(root -> childL, ans);
        ans.push_back(root -> val);
        traverse(root -> childR, ans);
    }

    Node* locate(int x){
        return locate(root, x);
    }

    Node* locate(Node *root, int x){
        if (root == nullptr) return nullptr;
        if (root -> val == x) return root;
        if (x < root -> val) return locate(root -> childL, x);
        else return locate(root -> childR, x);
    }

    bool exist(int x){
        return locate(x) != nullptr;
    }

    bool erase(int x){
        return erase(root, x);
    }

    bool erase(Node* &root, int x){
        if (root == nullptr) return false;
        if (x == root -> val) { // delete this shit
            if (root -> childR){ 
                int val = delete_left_most(root -> childR);
                root -> val = val;
            }
            else{
                Node* tmp = root;
                root = root -> childL;
                delete tmp;
            }
            return true;
        }
        if (x < root -> val) return erase(root -> childL, x);
        else return erase(root -> childR, x);
    }

    int delete_left_most(Node* &root){
        if (root -> childL) return delete_left_most(root -> childL);
        else if (root -> childR) return delete_left_most(root -> childR);
        int ans = root -> val;
        delete root;
        root = nullptr;
        return ans;
    }
};
