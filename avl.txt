struct Node{
    int val, depth;
    Node *childL, *childR;
    Node(int x = 0){depth = 1; val = x; childL = childR = nullptr;}

    int left_child_depth(){
        if (childL) return childL -> depth;
        return 0;
    }

    int right_child_depth(){
        if (childR) return childR -> depth;
        return 0;
    }
};  


struct AVLTree{
    Node* root;

    AVLTree(){
        root = nullptr;
    }

    void calculate_depth(Node* x){
        if (x == nullptr) return;
        x -> depth = std::max(x -> left_child_depth(), x -> right_child_depth()) + 1;
    }

    int unbalanced(Node* x){
        calculate_depth(x);
        int diff = (x -> left_child_depth()) - (x -> right_child_depth());
        if (diff >= 2) return -1;
        if (diff <= -2) return 1;
        return 0;
    }

    void insert(int x){
        insert(root, x);
    }

    void rotate_left(Node* &root){
        Node* tmp = root -> childL;
        root -> childL = root -> childL -> childR;
        tmp -> childR = root;
        calculate_depth(root);
        root = tmp;
        calculate_depth(root);
    }
    void rotate_right(Node* &root){
        Node* tmp = root -> childR;
        root -> childR = root -> childR -> childL;
        tmp -> childL = root;
        calculate_depth(root);
        root = tmp;
        calculate_depth(root);
    }


    void correct(Node* &root){
        if (unbalanced(root) == -1){ 
            rotate_left(root);
            if (unbalanced(root) == 1){
                rotate_left(root -> childR);
                calculate_depth(root);
                rotate_right(root);
            }
        }
        else if (unbalanced(root) == 1){ 
            rotate_right(root);
            if (unbalanced(root) == -1){
                rotate_right(root -> childL);
                calculate_depth(root);
                rotate_left(root);
            }
        }
    }

    void insert(Node* &root, int x){
        if (root == nullptr) {
            root = new Node(x);
            return;
        }
        if (x < root -> val) insert(root -> childL, x);
        else insert(root -> childR, x);

        correct(root);
        calculate_depth(root);
    }

    void traverse(){
        if (root){
            cout << root -> val;
            vector<int> ans;
            traverse(root, ans);
            cout << "\n";
            printArr(ans);
        }
    }

    void traverse(Node *root, vector<int> &ans){
        if (root -> childL) {
            cout << " -> " << root -> childL -> val;
            traverse(root -> childL, ans);
            cout << " -> " << root -> val;
        }

        ans.push_back(root -> val);

        if (root -> childR){
            cout << " -> " << root -> childR -> val;
            traverse(root -> childR, ans);
            cout << " -> " << root -> val;
        }
    }

    Node* locate(int x){
        return locate(root, x);
    }

    Node* locate(Node *root, int x){
        if (root == nullptr) return nullptr;
        if (root -> val == x) return root;
        if (x < root -> val) return locate(root -> childL, x);
        else return locate(root -> childR, x);
    }

    bool exist(int x){
        return locate(x) != nullptr;
    }

    bool check_correct_depth(){
        return check_correct_depth(root);
    }
    bool check_correct_depth(Node *root){
        if (root == nullptr) return true;
        int cur = root -> depth;
        calculate_depth(root);
        if (root -> depth != cur) return false;
        if (unbalanced(root)) return false;
        return check_correct_depth(root -> childL) && check_correct_depth(root -> childR);
    }

    bool erase(int x){
        return erase(root, x);
    }

    bool erase(Node* &root, int x){
        if (root == nullptr) return false;
        if (x == root -> val) { // delete this shit
            if (root -> childR){ 
                int val = delete_left_most(root -> childR);
                root -> val = val;
            }
            else{
                Node* tmp = root;
                root = root -> childL;
                delete tmp;
            }
            correct(root);
            return true;
        }
        bool check = false;
        if (x < root -> val) check = erase(root -> childL, x);
        else check = erase(root -> childR, x);
        correct(root);
        return check;
    }

    int delete_left_most(Node* &root){
        if (root -> childL) {
            int ans = delete_left_most(root -> childL);
            correct(root);
            return ans;
        } 
        else  {
            Node* tmp = root;
            root = root -> childR;
            int ans = tmp -> val;
            delete tmp;
            return ans;
        }
    }
};
